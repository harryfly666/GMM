# generate data using user-defined program
def generate_data(sigma, N, mean):
    X = np.zeros((N, 2))
    alpha_g = np.zeros((4, 1))
    label = np.zeros(N)
    for i in range(N):
        if np.random.random(1) < 0.1:
            X[i, :] = np.random.multivariate_normal(mean[0], sigma)
            alpha_g[0] += 1
            label[i] = 0
        elif 0.1 <= np.random.random(1) < 0.3:
            X[i, :] = np.random.multivariate_normal(mean[1], sigma)
            alpha_g[1] += 1
            label[i] = 1
        elif 0.3 <= np.random.random(1) < 0.6:
            X[i, :] = np.random.multivariate_normal(mean[2], sigma)
            alpha_g[2] += 1
            label[i] = 2
        else:
            X[i, :] = np.random.multivariate_normal(mean[3], sigma)
            alpha_g[3] += 1
            label[i] = 3
    alpha_g = alpha_g / N
    return X, alpha_g, label

# caculate the probability of a given data
def Gaussian(x, mean, sigma):
    a = np.exp(-0.5 * (x - mean) * sigma.I * np.transpose(x - mean))
    b = 2 * np.pi * (np.linalg.det(sigma) ** 0.5)
    return np.array(a/b).flatten()

# e step: update weights
def e_step(sigma, k, N):
    for i in range(N):
        denom = 0
        for j in range(k):
            denom += alpha[j] * Gaussian(np.matrix(X[i]), np.matrix(mean[j]), sigma)
        for j in range(k):
            numer = alpha[j] * Gaussian(np.matrix(X[i]), np.matrix(mean[j]), sigma)
            w[i, j] = numer / denom
    return w

# m step: update alpha and mean while supposing sigma is unchanged
def m_step(k, N):
    for j in range(k):
        denom = 0
        numer = 0 
        for i in range(N):
            numer += w[i, j] * X[i]
            denom += w[i, j]
        mean[j] = numer / denom
        alpha[j] = denom / N
    return mean, alpha
